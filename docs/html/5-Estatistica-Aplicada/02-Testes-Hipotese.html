Teste t para Uma Amostra

Objetivo: Comparar m√©dia da amostra com valor de refer√™ncia.

Exemplo: Biomassa m√©dia √© diferente de 250g?

from scipy import stats
import numpy as np

# Dados de biomassa (g)

# Valor de refer√™ncia (hipot√©tico)

# Realizar teste t para uma amostra

print("üß™ TESTE T PARA UMA AMOSTRA")
print("="*60)
print(f"\nH‚ÇÄ: Œº = {valor_referencia}g")
print(f"H‚ÇÅ: Œº ‚â† {valor_referencia}g")
print(f"\nM√©dia observada: {np.mean(biomassas):.2f}g")
print(f"Estat√≠stica t: {t_stat:.4f}")
print(f"P-valor: {p_valor:.4f}")

if p_valor < 0.05:
    print("\n‚úÖ Resultado: SIGNIFICATIVO (p < 0.05)")
    print("   Rejeitamos H‚ÇÄ: A m√©dia √© diferente de 250g")

Teste t para Amostras Independentes

Objetivo: Comparar m√©dias de dois grupos diferentes.

import pandas as pd

# Dados de duas esp√©cies
dados = {

                'Gracilaria', 'Gracilaria', 'Gracilaria', 'Gracilaria'],
    'biomassa_g': [245.3, 302.1, 275.4, 310.8, 295.7,
                   180.7, 195.8, 205.6, 220.4]

print("\n" + "="*60)
print("üß™ TESTE T PARA AMOSTRAS INDEPENDENTES")
print("="*60)
print(f"\nH‚ÇÄ: Œº‚ÇÅ = Œº‚ÇÇ (m√©dias iguais)")
print(f"H‚ÇÅ: Œº‚ÇÅ ‚â† Œº‚ÇÇ (m√©dias diferentes)")

print(f"\nüìä Estat√≠sticas descritivas:")
print(f"   Ulva:       n={len(ulva)}, m√©dia={ulva.mean():.2f}g ¬± {ulva.std():.2f}g")
print(f"   Gracilaria: n={len(gracilaria)}, m√©dia={gracilaria.mean():.2f}g ¬± {gracilaria.std():.2f}g")

print(f"\nüî¨ Resultado do teste:")
print(f"   Estat√≠stica t: {t_stat:.4f}")
print(f"   P-valor: {p_valor:.4f}")

if p_valor < 0.05:
    print("\n‚úÖ Conclus√£o: H√° diferen√ßa significativa entre as esp√©cies (p < 0.05)")
    diferenca = ulva.mean() - gracilaria.mean()
    print(f"   Ulva tem {abs(diferenca):.2f}g {'a mais' if diferenca > 0 else 'a menos'} que Gracilaria")
else:
    print("\n‚ùå Conclus√£o: N√£o h√° diferen√ßa significativa (p ‚â• 0.05)")

import numpy as np

# Dataset completo

    'biomassa_g': [245.3, 302.1, 275.4, 310.8, 295.7,  # Ingleses
                   180.7, 195.8, 205.6, 220.4, 198.5,  # Barra
                   310.2, 340.5, 315.2, 325.8, 330.1]  # Arma√ß√£o
}

df = pd.DataFrame(dados)

print("="*70)
print("üî¨ AN√ÅLISE COMPARATIVA COMPLETA")
print("="*70)

# ====================
# 1. ESTAT√çSTICAS DESCRITIVAS
# ====================

print("\nüìä ESTAT√çSTICAS POR PRAIA:")
print("-"*70)

for praia in df['praia'].unique():
    dados_praia = df[df['praia'] == praia]['biomassa_g']
    print(f"\n{praia}:")
    print(f"   n = {len(dados_praia)}")
    print(f"   M√©dia = {dados_praia.mean():.2f}g ¬± {dados_praia.std():.2f}g")
    print(f"   Mediana = {dados_praia.median():.2f}g")
    print(f"   Range = [{dados_praia.min():.2f}, {dados_praia.max():.2f}]g")

# ====================
# 2. TESTE DE NORMALIDADE
# ====================

print("\n" + "="*70)
print("üîç TESTE DE NORMALIDADE (SHAPIRO-WILK)")
print("="*70)

for praia in df['praia'].unique():
    dados_praia = df[df['praia'] == praia]['biomassa_g']
    stat, p_valor = stats.shapiro(dados_praia)

    normal = "SIM ‚úÖ" if p_valor > 0.05 else "N√ÉO ‚ùå"
    print(f"\n{praia}: p={p_valor:.4f} ‚Üí Distribu√ß√£o normal? {normal}")

# ====================
# 3. ANOVA
# ====================

print("\n" + "="*70)
print("üß™ ANOVA ONE-WAY")
print("="*70)

ingleses = df[df['praia'] == 'Ingleses']['biomassa_g']
barra = df[df['praia'] == 'Barra']['biomassa_g']
armacao = df[df['praia'] == 'Arma√ß√£o']['biomassa_g']

f_stat, p_valor = stats.f_oneway(ingleses, barra, armacao)

print(f"\nH‚ÇÄ: Œº‚ÇÅ = Œº‚ÇÇ = Œº‚ÇÉ")
print(f"Estat√≠stica F: {f_stat:.4f}")
print(f"P-valor: {p_valor:.6f}")

if p_valor < 0.05:
    print("\n‚úÖ RESULTADO: H√° diferen√ßa significativa entre praias (p < 0.05)")
else:
    print("\n‚ùå RESULTADO: N√£o h√° diferen√ßa significativa (p ‚â• 0.05)")

# ====================
# 4. COMPARA√á√ïES PAREADAS (Post-hoc)
# ====================

if p_valor < 0.05:
    print("\n" + "="*70)
    print("üìä COMPARA√á√ïES PAREADAS (TESTE T)")
    print("="*70)

    comparacoes = [
        ('Ingleses', ingleses, 'Barra', barra),
        ('Ingleses', ingleses, 'Arma√ß√£o', armacao),
        ('Barra', barra, 'Arma√ß√£o', armacao)
    ]

    for nome1, dados1, nome2, dados2 in comparacoes:
        t_stat, p = stats.ttest_ind(dados1, dados2)

        # Corre√ß√£o de Bonferroni
        p_corrigido = p * 3  # 3 compara√ß√µes

        sig = "‚úÖ SIG" if p_corrigido < 0.05 else "‚ùå NS"
        print(f"\n{nome1} vs {nome2}:")
        print(f"   p-valor: {p:.4f} (corrigido: {p_corrigido:.4f}) {sig}")
