üîç Identifica√ß√£o de Outliers

M√©todo IQR: Valores fora de 1.5 √ó IQR s√£o considerados outliers.

def identificar_outliers(dados):
    """Identifica outliers usando m√©todo IQR"""
    q1 = np.percentile(dados, 25)
    q3 = np.percentile(dados, 75)
    iqr = q3 - q1

    # Limites
    limite_inferior = q1 - 1.5 * iqr
    limite_superior = q3 + 1.5 * iqr

    # Identificar outliers
    outliers = [x for x in dados if x < limite_inferior or x > limite_superior]

    print(f"\nüîç DETEC√á√ÉO DE OUTLIERS:")
    print(f"   Limite inferior: {limite_inferior:.2f}")
    print(f"   Limite superior: {limite_superior:.2f}")

    if outliers:
        print(f"   ‚ö†Ô∏è {len(outliers)} outlier(s) detectado(s): {outliers}")
    else:
        print(f"   ‚úÖ Nenhum outlier detectado")

    return outliers
