def gerar_dados_coletas(n=50):
    """Gera dados sint√©ticos de coletas"""

    praias = ['Ingleses', 'Barra da Lagoa', 'Arma√ß√£o', 'Garopaba', 'Laguna']

            'praia': praia,

    return coletas
